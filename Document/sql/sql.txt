CREATE TABLE t_cam_datasets (
id int(11) NOT NULL AUTO_INCREMENT, # '主键',
`name` varchar(32) NOT NULL, # '数据名称',
frames int(11) DEFAULT 0, # ‘数据总量’,
remarks varchar(200) DEFAULT NULL , # '备注',
recorder varchar(32) not null,
record_time varchar(32) not null,
record_place varchar(32) not null,
create_person varchar(32) NOT NULL , # '创建人员',
create_time varchar(32) NOT NULL , # '创建时间',
purpose varchar(32),#创建目的
update_person varchar(32) not null,
update_time varchar(32) not null,
path varchar(32) not null,
PRIMARY KEY (id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

right_point_x int(11) DEFAULT NULL , # ‘右下X坐标’,
